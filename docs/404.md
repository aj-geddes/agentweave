// Smooth scroll to browse sections
document.addEventListener('DOMContentLoaded', function() {
  const browseBelowLink = document.querySelector('a[href="#browse-sections"]');
  if (browseBelowLink) {
    browseBelowLink.addEventListener('click', function(e) {
      e.preventDefault();
      const target = document.getElementById('browse-sections');
      if (target) {
        target.scrollIntoView({ behavior: 'smooth', block: 'start' });
      }
    });
  }
});
